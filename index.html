fetch("datos_filtrados.json") .then(response => response.json()) .then(data => { const select = document.getElementById("continenteSelect"); function getContinente(pais) { const mapa = { "Chile": "América", "Argentina": "América", "Brasil": "América", "Estados Unidos": "América", "Canadá": "América", "Alemania": "Europa", "Francia": "Europa", "España": "Europa", "Italia": "Europa", "China": "Asia", "India": "Asia", "Japón": "Asia", "Australia": "Oceanía", "Sudáfrica": "África", "Egipto": "África" }; return mapa[pais] || "Otro"; } data.forEach(d => { d.Continente = getContinente(d["País"]); }); function actualizarVista(continente) { let filtrado = data; if (continente !== "Todos") { filtrado = data.filter( d => d.Continente === continente ); } crearGrafico(filtrado); crearRanking(filtrado); } function crearGrafico(datos) { const x = datos.map(d => d["País"]); const y = datos.map( d => d["Kilometros cargados con 50 dolares"] ); const colores = y.map(valor => valor >= 800 ? "#16a34a" : "#dc2626" ); const trace = { x: x, y: y, type: "bar", marker: { color: colores } }; const layout = { title: "Kilómetros recorridos con 50 USD", xaxis: { title: "País" }, yaxis: { title: "Kilómetros" } }; Plotly.newPlot("graficoBarras", [trace], layout); } function crearRanking(datos) { const ordenados = [...datos].sort( (a, b) => b["Kilometros cargados con 50 dolares"] - a["Kilometros cargados con 50 dolares"] ); const mejores = ordenados.slice(0, 5); const peores = ordenados.slice(-5).reverse(); document.getElementById("topMejores").innerHTML = mejores .map(d => `
  • ${d["País"]}: ${d["Kilometros cargados con 50 dolares"].toFixed(1)} km
    • ` ) .join(""); document.getElementById("topPeores").innerHTML = peores .map(d => `
  • ${d["País"]}: ${d["Kilometros cargados con 50 dolares"].toFixed(1)} km
    • ` ) .join(""); } select.addEventListener("change", e => { actualizarVista(e.target.value); }); actualizarVista("Todos"); });
